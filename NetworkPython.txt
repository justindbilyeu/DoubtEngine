G = nx.Graph()
for n,t in nodes.items(): G.add_node(n, turn=t)
G.add_edges_from(edges)

# ---- METRICS ----
cent = nx.degree_centrality(G)
deg = dict(G.degree())
turn = nodes

# forward reach: how far into the future does each node's influence extend
reach = {}
for n in G.nodes():
    spans = [turn[m]-turn[n] for m in G.neighbors(n)]
    reach[n] = max(spans) if spans else 0

# novelty per turn (conceptual "surprise": new nodes introduced each turn)
T = max(turn.values())
novelty = [sum(1 for v in turn.values() if v==t) for t in range(1,T+1)]
# recurrence: edges each turn that point BACK to earlier turns (reuse of prior structure)
back_edges = [0]*(T+1)
for u,v in G.edges():
    hi=max(turn[u],turn[v]); lo=min(turn[u],turn[v])
    if hi!=lo: back_edges[hi]+=1
backseq=[back_edges[t] for t in range(1,T+1)]

print("=== TOP HUBS (degree centrality) ===")
for n,c in sorted(cent.items(),key=lambda x:-x[1])[:9]:
    print(f"  {deg[n]:>2} links  c={c:.3f}  turn {turn[n]:>2}  {n.replace(chr(10),' ')}")

print("\n=== TOP FORWARD-REACH (seeds the future) ===")
for n,r in sorted(reach.items(),key=lambda x:-x[1])[:9]:
    print(f"  reach +{r:>2} turns  (born T{turn[n]:>2})  {n.replace(chr(10),' ')}")

print("\n=== NOVELTY PER TURN (new concepts introduced) ===")
print("  turn :", " ".join(f"{t:>2}" for t in range(1,T+1)))
print("  new  :", " ".join(f"{n:>2}" for n in novelty))
print("  back :", " ".join(f"{b:>2}" for b in backseq), " (edges reusing earlier structure)")

pos = nx.spring_layout(G, k=0.85, iterations=300, seed=7)
turns = np.array([turn[n] for n in G.nodes()])
cents = np.array([cent[n] for n in G.nodes()])
cmap = plt.cm.turbo
norm = mpl.colors.Normalize(vmin=1, vmax=T)

# edges colored by temporal span: long-range forward edges salient, local edges faint
for u,v in G.edges():
    span = abs(turn[u]-turn[v])
    if span>=4:
        ax.plot([pos[u][0],pos[v][0]],[pos[u][1],pos[v][1]], color='#d62728', lw=1.4, alpha=0.55, zorder=1)
    else:
        ax.plot([pos[u][0],pos[v][0]],[pos[u][1],pos[v][1]], color='#9aa0a6', lw=0.6, alpha=0.30, zorder=1)

sizes = 250 + 7000*cents
ax.scatter([pos[n][0] for n in G.nodes()],[pos[n][1] for n in G.nodes()],
           s=sizes, c=turns, cmap=cmap, norm=norm, edgecolors='white', linewidths=1.2, zorder=2)

# label the most central nodes prominently; small font for the rest
ranked = sorted(cent.items(), key=lambda x:-x[1])
hub_set = set(n for n,_ in ranked[:17])
for n in G.nodes():
    x,y = pos[n]
    if n in hub_set:
        fs = 8.5 + 22*cent[n]
        ax.text(x,y,n, fontsize=min(fs,11.5), ha='center', va='center', zorder=3,

        ax.text(x,y+0.045,n, fontsize=6.2, ha='center', va='center', zorder=3, color='#202124', alpha=0.85)
